11pub(crate) fn run_dataflow(r: RelExpr) -> Vec<Vec<Datum>> {
12    let mut df = Dfir::new();
13
14    let output_port = render_relational(&mut df, r);
15
16    let output = Rc::new(RefCell::new(Vec::new()));
17    let inner = output.clone();
18
19    df.add_subgraph_sink("output sink", output_port, move |_ctx, recv| {
20        for v in recv.take_inner() {
21            (*inner).borrow_mut().push(v);
22        }
23    });
24
25    df.run_available();
26
27    let v = (*output).borrow();
28    v.clone()
29}

31fn render_relational(df: &mut Dfir, r: RelExpr) -> RecvPort<VecHandoff<Vec<Datum>>> {
32    let (send_port, recv_port) = df.make_edge("handoff");
33    match r {
34        RelExpr::Values(mut v) => {
35            // TODO: drip-feed data?
36            let scope = Vec::new();
37            df.add_subgraph_source("value source", send_port, move |_ctx, send| {
38                send.give(Iter(
39                    v.drain(..)
40                        .map(|row| row.into_iter().map(|e| e.eval(&scope)).collect()),
41                ));
42            });
43        }
44        RelExpr::Filter(preds, v) => {
45            let input = render_relational(df, *v);
46            df.add_subgraph_in_out("filter", input, send_port, move |_ctx, recv, send| {
47                send.give(Iter(recv.take_inner().into_iter().filter(|row| {
48                    preds.iter().all(|p| p.eval(row) == Datum::Bool(true))
49                })));
50            });
51        }
52        RelExpr::Project(exprs, v) => {
53            let input = render_relational(df, *v);
54            df.add_subgraph_in_out("project", input, send_port, move |_ctx, recv, send| {
55                send.give(Iter(
56                    recv.take_inner()
57                        .into_iter()
58                        .map(|row| exprs.iter().map(|e| e.eval(&row)).collect()),
59                ));
60            });
61        }
62    }
63    recv_port
64}
