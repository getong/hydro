8#[stageleft::entry]
9fn raise_to_power(_ctx: BorrowBounds<'_>, value: RuntimeData<i32>, power: u32) -> impl Quoted<i32> {
10    if power == 1 {
11        q!(value).boxed()
12    } else if power % 2 == 0 {
13        let half_result = raise_to_power(_ctx, value, power / 2);
14        q!({
15            let v = half_result;
16            v * v
17        })
18        .boxed()
19    } else {
20        let half_result = raise_to_power(_ctx, value, power / 2);
21        q!({
22            let v = half_result;
23            (v * v) * value
24        })
25        .boxed()
26    }
27}
